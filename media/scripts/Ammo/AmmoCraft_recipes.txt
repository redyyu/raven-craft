    /***** Gunpowder *****/

    /* Override Vanilla recipe */

    /* Gatherpowder is calculated with Gunpowder usage */
    recipe Gather Gunpowder {
        Bullets38/Bullets45/Bullets9mm/223Bullets/308Bullets/Bullets44/556Bullets/ShotgunShells=50,

        Result:GunPowder,
        Time:100.0,
        Override:true,
    }

    /* Craft Gunpowder */
    recipe Craft Gunpowder {
        keep MortarPestle,
        Fertilizer=2,
        Charcoal=1,
        [Recipe.GetItemTypes.Sugar];2,

        Prop1: Source=1,
        Sound: Plastering,
        Result: GunPowder,
        Time: 300.0,
        NeedToBeLearn: true,
        SkillRequired: Doctor=2;Farming=8;Woodwork=6,
        Category:Ammo,
    }

    recipe Craft Gunpowder {
